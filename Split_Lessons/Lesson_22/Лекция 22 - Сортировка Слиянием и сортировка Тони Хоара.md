```python
def merge_sort(a: list):  
    """–°–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞ —Å–ª–∏—è–Ω–∏–µ–º.–í—Å–µ–≥–¥–∞ O(n*logn)"""  
    if len(a) <= 1:

    mid = len(a) // 2  

    left = merge_sort(a[:mid])  # –¥–µ–ª–∏–º –Ω–∞ 2 —á–∞—Å—Ç–∏ 
    right = merge_sort(a[mid:]) # –ª–µ–≤—É—é –∏ –ø—Ä–∞–≤—É—é 
      
    return merge(left, right)  
  
def merge(left: list, right: list):  
    c = []  
    i = j = 0  
    while i < len(left) and j < len(right):  
        if left[i] < right[j]:  
            c.append(left[i])  
            i += 1  
        else:  
            c.append(right[j])  
            j += 1  
  
    c.extend(left[i:])  
    c.extend(right[j:])  
    return c
```
