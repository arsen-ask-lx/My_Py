from typing import List

def longest_common_subsequence(a: List[int], b: List[int]):  
    """–ë—É–¥–µ–º –∏—Å–∫–∞—Ç—å —Å–∞–º—É—é –¥–ª–∏–Ω–Ω—É—é –ø–æ–¥–ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç—å"""  
    c = [[0 for i in range(len(b) + 1)] for j in range(len(a) + 1)]  
    for i in range(1, len(a) + 1):  
        for j in range(1, len(b) + 1):  
            if a[i-1] == b[j-1]:  
                c[i][j] = 1 + c[i - 1][j - 1]  
            else:  
                c[i][j] = max(c[i - 1][j], c[i][j - 1])  
  
    n = len(a)  
    m = len(b)  
    result = []  
    while n > 0 and m > 0:  
        if a[n-1] == b[m-1]:  
            result.append(a[n-1])  
            n -= 1  
            m -= 1  
        elif c[n - 1][m] > c[n][m - 1]:  
            n -= 1  
        else:  
            m -= 1  
    result.reverse()  
  
    return c[-1][-1], result  
  
  
print(longest_common_subsequence([8, 4, 3, 5, 7], [3, 5, 4, 7]))
