```python
from typing import List  
from pprint import pprint  
  
  
def longest_increasing_subsequence(a: List[int]):  
    """–ë—É–¥–µ–º –∏—Å–∫–∞—Ç—å —Å–∞–º—É—é –¥–ª–∏–Ω–Ω—É—é –Ω–∞–∏–±–æ–ª—å—à—É—é –ø–æ–¥–ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç—å"""  
    n = len(a)  
    prev = [-1] * n  
    subseq_len = [1] * n  
    for i in range(1, len(a)):  
        for j in range(i):  
            if a[i] > a[j] and subseq_len[i] < subseq_len[j] + 1:  
                subseq_len[i] = subseq_len[j] + 1  
                prev[i] = j  
  
    max_value = max(a)  
    max_index = a.index(max_value)  
  
    result = []  
    k = max_index  
    while k != -1:  
        result.append(a[k])  
        k = prev[k]  
    result.reverse()  
      
    return result  
  
print(longest_increasing_subsequence([1,5,1,2,0,9,11]))
```
